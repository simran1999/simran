class Movies implements Comparable<Movies> {
    int year;
    int rating;
    String moviename;
    Movies(int year,int rating,String moviename){
        this.year=year;
        this.rating=rating;
        this.moviename=moviename;
    }

    
    public int compareTo(Movies t) {
        return this.year-t.year;
    }
